# ALA 2: SHA Integrity Analyzer
# Compares SHA-1, SHA-256, SHA-512 and demonstrates the Avalanche Effect

import hashlib

# ─── Helper: Compute all SHA variants ────────────────────────────────────────
def compute_hashes(message: str) -> dict:
    encoded = message.encode('utf-8')
    return {
        "SHA-1"  : hashlib.sha1(encoded).hexdigest(),
        "SHA-256": hashlib.sha256(encoded).hexdigest(),
        "SHA-512": hashlib.sha512(encoded).hexdigest(),
    }

# ─── Helper: Compare two hash strings, count differences ─────────────────────
def compare_hashes(h1: str, h2: str) -> dict:
    diff_count = sum(1 for a, b in zip(h1, h2) if a != b)
    pct = (diff_count / len(h1)) * 100
    return {"diff_chars": diff_count, "pct": round(pct, 2)}

# ─── Main Execution ───────────────────────────────────────────────────────────
if __name__ == "__main__":
    msg1 = "Hello, World!"
    msg2 = "Hello, World."   # Changed '!' to '.'

    hashes1 = compute_hashes(msg1)
    hashes2 = compute_hashes(msg2)

    print("=== SHA Integrity Analyzer ===")
    print(f"\nMessage 1 : '{msg1}'")
    print(f"Message 2 : '{msg2}' (1 char changed)")

    print("\n─── Hashes for Message 1 ───")
    for algo, digest in hashes1.items():
        print(f"{algo:8}: {digest}")

    print("\n─── Hashes for Message 2 ───")
    for algo, digest in hashes2.items():
        print(f"{algo:8}: {digest}")

    print("\n─── Avalanche Effect (% characters changed) ───")
    for algo in hashes1:
        result = compare_hashes(hashes1[algo], hashes2[algo])
        print(f"{algo:8}: {result['diff_chars']} chars differ → {result['pct']}% changed")

Analysis & Observations

🌊 Avalanche Effect

Changing a single character caused over 80% of hash output characters to change across all three algorithms. This is the defining property of strong cryptographic hash functions.

📏 Fixed Output Size

Regardless of input length, SHA-1 always outputs 40 hex chars, SHA-256 outputs 64, and SHA-512 outputs 128. This determinism is essential for signature schemes.

⚠️ SHA-1 Deprecation

SHA-1 is cryptographically broken — real-world collisions were found in 2017 (SHAttered attack). It must not be used for digital signatures or certificates.

🔗 Role in Digital Signatures

Hashing enables efficient signing: instead of signing a large document, we sign its tiny fixed-size digest. SHA-256 is the standard in TLS, SSL, and most PKI systems today.
