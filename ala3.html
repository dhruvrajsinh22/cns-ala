Objective

Design and simulate a Message Authentication Code (MAC) mechanism using HMAC in Python. A sender generates a MAC for a message using a shared secret key; the receiver verifies the MAC to confirm both data integrity (message not altered) and authentication (message came from legitimate sender).

Sender ↔ Receiver Protocol

📤
Sender
1 Has plaintext message to send
2 Uses shared secret key + HMAC-SHA256 to compute MAC
3 Transmits message + MAC to receiver
→

📥
Receiver
1 Receives message + MAC
2 Re-computes MAC using same secret key + HMAC-SHA256
3 Compares MACs — if equal, message is authentic & intact

Source Code

sender.py PYTHON 
# ALA 3: MAC-Based Message Authentication — Sender Script
# Generates HMAC for a message using a shared secret key

import hmac
import hashlib
import json

# ─── Shared Secret Key (known to both sender & receiver) ──────────────────────
SECRET_KEY = b"supersecretkey_ALA3_2024"

def generate_mac(message: str, key: bytes) -> str:
    """Generate HMAC-SHA256 for the given message."""
    mac = hmac.new(key, message.encode('utf-8'), hashlib.sha256)
    return mac.hexdigest()

if __name__ == "__main__":
    message = "Transfer ₹5000 to Account 9876543210"

    mac_value = generate_mac(message, SECRET_KEY)

    # Package message and MAC together (simulating network transmission)
    packet = {"message": message, "mac": mac_value}

    print("=== SENDER ===")
    print(f"\n[*] Original Message : {message}")
    print(f"[+] HMAC-SHA256 MAC  : {mac_value}")
    print(f"\n[→] Packet transmitted:\n    {json.dumps(packet, indent=4)}")

    # Save to file to simulate network transfer
    with open("packet.json", "w") as f:
        json.dump(packet, f)
receiver.py PYTHON 
# ALA 3: MAC-Based Message Authentication — Receiver Script
# Verifies the HMAC to confirm message integrity and authenticity

import hmac
import hashlib
import json

# ─── Shared Secret Key (must match sender's key) ──────────────────────────────
SECRET_KEY = b"supersecretkey_ALA3_2024"

def verify_mac(message: str, received_mac: str, key: bytes) -> bool:
    """Re-compute MAC and compare using constant-time comparison."""
    expected_mac = hmac.new(key, message.encode('utf-8'), hashlib.sha256).hexdigest()
    # hmac.compare_digest prevents timing attacks
    return hmac.compare_digest(expected_mac, received_mac)

if __name__ == "__main__":
    # Receive packet (simulated from file)
    with open("packet.json", "r") as f:
        packet = json.load(f)

    message      = packet["message"]
    received_mac = packet["mac"]

    print("=== RECEIVER ===")
    print(f"\n[←] Received Message : {message}")
    print(f"[←] Received MAC     : {received_mac}")

    # Test 1: Verify unmodified message
    is_valid = verify_mac(message, received_mac, SECRET_KEY)
    print(f"\n[✓] MAC Verification (original) : {is_valid}")

    # Test 2: Tampered message
    tampered_msg = "Transfer ₹50000 to Account 1111111111"
    is_tampered  = verify_mac(tampered_msg, received_mac, SECRET_KEY)
    print(f"[✗] MAC Verification (tampered)  : {is_tampered}")

    if is_valid:
        print("\n[SECURE] Message is authentic and unmodified!")
    else:
        print("\n[ALERT]  Message has been tampered! Discard it.")

Analysis

🔑 Shared Key Security

HMAC's security relies on the secrecy of the shared key. If the key is compromised, an attacker can forge valid MACs. Secure key exchange (e.g., Diffie-Hellman) is essential.

⏱️ Timing Attack Prevention

hmac.compare_digest() uses constant-time comparison, preventing timing side-channel attacks where an attacker could guess the MAC character by character.

💼 Real-World Use

HMAC is used in JWT tokens, REST API authentication, TLS record authentication, and WhatsApp/Signal message verification — billions of messages daily.

🛡️ Data Integrity

The tampered message test proves integrity: even changing a single digit (₹5000 → ₹50000) causes a completely different HMAC, immediately detected by the receiver.
